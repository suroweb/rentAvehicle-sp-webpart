```typescript
/**
 * Zod schema for validating timezone update requests.
 * Validates that the timezone string is a valid IANA identifier by attempting
 * to construct an Intl.DateTimeFormat with it (throws RangeError for invalid timezones).
 */
export const TimezoneUpdateSchema = z.object({
  timezone: z.string().min(1).max(64).refine(
    (tz: string): boolean => {
      try {
        new Intl.DateTimeFormat(undefined, { timeZone: tz });
        return true;
      } catch {
        return false;
      }
    },
    { message: 'Invalid IANA timezone identifier' }
  ),
});
```

```typescript
/**
 * PATCH /api/backoffice/locations/{id}/timezone
 * Update a location's IANA timezone.
 * Requires Admin or SuperAdmin role.
 */
async function updateLocationTimezone(
  request: HttpRequest,
  context: InvocationContext
): Promise<HttpResponseInit> {
  try {
    const user = await getUserFromRequest(request);
    if (!user) {
      return { status: 401, jsonBody: { error: 'Not authenticated' } };
    }
    if (!isAdminOrSuperAdmin(user)) {
      return {
        status: 403,
        jsonBody: { error: 'Admin or SuperAdmin role required' },
      };
    }

    const id = parseInt(request.params.id, 10);
    if (isNaN(id)) {
      return { status: 400, jsonBody: { error: 'Invalid location ID' } };
    }

    const body = (await request.json()) as Record<string, unknown>;
    const parsed = TimezoneUpdateSchema.safeParse(body);
    if (!parsed.success) {
      return {
        status: 400,
        jsonBody: { error: 'Validation failed', details: parsed.error.flatten() },
      };
    }

    const updated = await updateTimezone(id, parsed.data.timezone);
    if (!updated) {
      return { status: 404, jsonBody: { error: 'Location not found' } };
    }

    return { jsonBody: { success: true } };
  } catch (error) {
    context.error('updateLocationTimezone failed:', error);
    return {
      status: 500,
      jsonBody: { error: 'Internal server error' },
    };
  }
}
```

IMPORTANT: Use a Node.js script to generate the full list. Do NOT hand-write or guess entries. The script should:
1. Call `Intl.supportedValuesOf('timeZone')` to get all valid IANA zones
2. For each zone, format the UTC offset as `(UTC+HH:MM)` or `(UTC-HH:MM)`
3. Sort by offset then alphabetically
4. Output the full TypeScript file

Run the generation script inline:
```bash
node -e "
const zones = Intl.supportedValuesOf('timeZone');
const jan = new Date(2026, 0, 15, 12, 0, 0);

function getOffset(tz) {
  const f = new Intl.DateTimeFormat('en-US', { timeZone: tz, timeZoneName: 'shortOffset' });
  const parts = f.formatToParts(jan);
  const tzPart = parts.find(p => p.type === 'timeZoneName');
  if (!tzPart) return { label: '(UTC+00:00)', sort: 0 };
  const raw = tzPart.value.replace('GMT', '');
  if (!raw) return { label: '(UTC+00:00)', sort: 0 };
  const sign = raw.charAt(0);
  const nums = raw.substring(1).split(':');
  const hours = nums[0].padStart(2, '0');
  const mins = nums.length > 1 ? nums[1].padStart(2, '0') : '00';
  const sortVal = (sign === '-' ? -1 : 1) * (parseInt(hours) * 60 + parseInt(mins));
  return { label: '(UTC' + sign + hours + ':' + mins + ')', sort: sortVal };
}

const entries = zones.map(tz => {
  const o = getOffset(tz);
  return { key: tz, text: o.label + ' ' + tz, sort: o.sort };
});

entries.sort((a, b) => a.sort - b.sort || a.key.localeCompare(b.key));

const lines = entries.map(e => '  { key: ' + JSON.stringify(e.key) + ', text: ' + JSON.stringify(e.text) + ' },');
console.log(lines.join('\n'));
" > /tmp/tz_entries.txt
```
