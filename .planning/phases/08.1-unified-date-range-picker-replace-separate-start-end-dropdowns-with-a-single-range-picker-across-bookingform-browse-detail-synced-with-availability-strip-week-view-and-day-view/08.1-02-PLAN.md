3. **Compute overlay bounds for the visible week:**

Create a `computeStripOverlay` function that maps the range to visible strip columns:

```typescript
function computeStripOverlay(
  range: IRangeState,
  dayColumns: Array<{ dayDate: Date; hourBlocks: IHourBlock[] }>,
  stripStartHour: number,
  stripEndHour: number
): { startDayIdx: number; startHourIdx: number; endDayIdx: number; endHourIdx: number } | null {
  // For each day column, check if the range overlaps this day
  // Clamp range start/end to the visible week's first and last day
  // Clamp range start/end hours to STRIP_START_HOUR and stripEndHour
  // Return null if range is entirely outside the visible week

  const firstVisibleDay = dayColumns[0].dayDate;
  const lastVisibleDay = dayColumns[dayColumns.length - 1].dayDate;

  // Range start/end at midnight for day comparison
  const rangeStartDay = new Date(range.startDate.getFullYear(), range.startDate.getMonth(), range.startDate.getDate());
  const rangeEndDay = new Date(range.endDate.getFullYear(), range.endDate.getMonth(), range.endDate.getDate());
  const firstDay = new Date(firstVisibleDay.getFullYear(), firstVisibleDay.getMonth(), firstVisibleDay.getDate());
  const lastDay = new Date(lastVisibleDay.getFullYear(), lastVisibleDay.getMonth(), lastVisibleDay.getDate());

  // No overlap: range is entirely before or after visible week
  if (rangeEndDay < firstDay || rangeStartDay > lastDay) return null;

  // Clamp to visible boundaries
  const clampedStartDay = rangeStartDay < firstDay ? firstDay : rangeStartDay;
  const clampedEndDay = rangeEndDay > lastDay ? lastDay : rangeEndDay;

  // Find day indices
  let startDayIdx = -1;
  let endDayIdx = -1;
  for (let i = 0; i < dayColumns.length; i++) {
    const colDay = dayColumns[i].dayDate;
    if (colDay.getTime() === clampedStartDay.getTime()) startDayIdx = i;
    if (colDay.getTime() === clampedEndDay.getTime()) endDayIdx = i;
  }
  if (startDayIdx === -1 || endDayIdx === -1) return null;

  // Clamp hours
  const clampedStartHour = rangeStartDay < firstDay ? stripStartHour : Math.max(range.startHour, stripStartHour);
  const clampedEndHour = rangeEndDay > lastDay ? stripEndHour : Math.min(range.endHour, stripEndHour);

  return {
    startDayIdx,
    startHourIdx: clampedStartHour - stripStartHour,
    endDayIdx,
    endHourIdx: clampedEndHour - stripStartHour,
  };
}
```

3. **Implement `handleDragPointerMove`:**

```typescript
const handleDragPointerMove = React.useCallback(function onDragMove(
  e: React.PointerEvent<HTMLDivElement>
): void {
  if (!draggingEdge.current) return;

  // Find which day column we're over by checking each column's bounding rect
  const stripGrid = (e.currentTarget as HTMLElement).closest('.' + styles.stripBlocks);
  if (!stripGrid) return;

  // Get the bounding rect of the blocks container
  const rect = stripGrid.getBoundingClientRect();
  const relativeY = e.clientY - rect.top;
  const blockHeight = rect.height / (mobileEndHour - STRIP_START_HOUR);
  const hourIndex = Math.floor(relativeY / blockHeight);
  const targetHour = Math.max(STRIP_START_HOUR, Math.min(mobileEndHour - 1, hourIndex + STRIP_START_HOUR));

  // Find which day column the pointer is in
  // For now, drag handles only adjust hours within their day column
  // Multi-day dragging would navigate the strip -- keep it simple: adjust hour only

  // Snap-to-boundary: check if targetHour is a booked slot
  const currentDayCol = findDayColumnForElement(e.currentTarget as HTMLElement);
  if (currentDayCol === undefined) return;

  const col = dayColumns[currentDayCol];
  const adjustedHour = snapToBoundary(targetHour, col.hourBlocks, draggingEdge.current);

  if (draggingEdge.current === 'start') {
    // Ensure start doesn't go past end
    if (adjustedHour < range.endHour || currentDayCol < findDayIndex(range.endDate)) {
      onRangeChange({ startDate: col.dayDate, startHour: adjustedHour });
    }
  } else {
    // Ensure end doesn't go before start
    const endHourValue = adjustedHour >= 23 ? 23 : adjustedHour + 1;
    if (endHourValue > range.startHour || currentDayCol > findDayIndex(range.startDate)) {
      onRangeChange({ endDate: col.dayDate, endHour: endHourValue });
    }
  }
}, [dayColumns, range, onRangeChange, mobileEndHour]);
```

5. **Implement `snapToBoundary` helper:**

```typescript
function snapToBoundary(
  targetHour: number,
  hourBlocks: IHourBlock[],
  edge: 'start' | 'end'
): number {
  // Find the block at targetHour offset from STRIP_START_HOUR
  const blockIndex = targetHour - STRIP_START_HOUR;
  if (blockIndex < 0 || blockIndex >= hourBlocks.length) return targetHour;

  const block = hourBlocks[blockIndex];
  if (!block.isBooked && !block.isPast) return targetHour;

  // Target is booked/past -- find nearest free boundary
  if (edge === 'end') {
    // Walk backward to find last free hour before the booked block
    for (let i = blockIndex - 1; i >= 0; i--) {
      if (!hourBlocks[i].isBooked && !hourBlocks[i].isPast) {
        return hourBlocks[i].hour;
      }
    }
    return STRIP_START_HOUR;
  } else {
    // Walk forward to find first free hour after the booked block
    for (let i = blockIndex + 1; i < hourBlocks.length; i++) {
      if (!hourBlocks[i].isBooked && !hourBlocks[i].isPast) {
        return hourBlocks[i].hour;
      }
    }
    return hourBlocks[hourBlocks.length - 1].hour;
  }
}
```
