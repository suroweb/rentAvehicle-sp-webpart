5. **Implement `handleDayViewDragMove`:**

Map pointer X position to hour column:
```typescript
const handleDayViewDragMove = React.useCallback(function onDragMove(
  e: React.PointerEvent<HTMLDivElement>
): void {
  if (!draggingEdge.current) return;

  // Find the timeline grid element
  const grid = (e.currentTarget as HTMLElement).closest('.' + styles.timelineGrid);
  if (!grid) return;

  const gridRect = grid.getBoundingClientRect();
  // First column is vehicle labels -- skip it
  // Get column width: (gridRect.width - labelColumnWidth) / TOTAL_HOUR_COLUMNS
  const labelWidth = gridRect.width / (TOTAL_HOUR_COLUMNS + 1); // approximate
  const hourAreaWidth = gridRect.width - labelWidth;
  const relativeX = e.clientX - gridRect.left - labelWidth;
  const hourIndex = Math.floor(relativeX / (hourAreaWidth / TOTAL_HOUR_COLUMNS));
  const targetHour = Math.max(TIMELINE_START_HOUR, Math.min(TIMELINE_END_HOUR - 1, hourIndex + TIMELINE_START_HOUR));

  // Snap to boundary: check if targetHour is booked for ANY vehicle (conservative -- prevents overlap)
  // Actually, per the single-vehicle scope of bookings, check the current vehicle's bookings
  // For simplicity, just check against the bookedHours set
  const snappedHour = snapToFreeHour(targetHour, draggingEdge.current);

  if (draggingEdge.current === 'start') {
    if (snappedHour < range.endHour) {
      onRangeChange({ startHour: snappedHour });
    }
  } else {
    const endVal = snappedHour >= 23 ? 23 : snappedHour + 1;
    if (endVal > range.startHour) {
      onRangeChange({ endHour: endVal });
    }
  }
}, [range, onRangeChange]);
```

6. **Snap-to-boundary for Day View:**

Build a `bookedHoursSet` from the loaded timeline bookings for the current vehicle and check against it:
```typescript
function snapToFreeHour(targetHour: number, edge: 'start' | 'end'): number {
  // If the hour is free, return it
  if (!allBookedHoursSet.has(targetHour)) return targetHour;
  // Walk in the appropriate direction to find the nearest free hour
  if (edge === 'end') {
    for (let h = targetHour - 1; h >= TIMELINE_START_HOUR; h--) {
      if (!allBookedHoursSet.has(h)) return h;
    }
    return TIMELINE_START_HOUR;
  } else {
    for (let h = targetHour + 1; h < TIMELINE_END_HOUR; h++) {
      if (!allBookedHoursSet.has(h)) return h;
    }
    return TIMELINE_END_HOUR - 1;
  }
}
```

Note: In Day View, since multiple vehicles are shown, build the booked hours set for ALL vehicles on the timeline for a conservative approach, OR for the vehicle that the current booking form targets (vehicleId). Since AvailabilityTimeline shows all vehicles at the location, use the booked hours for the vehicle matching the form's vehicleId (passed from VehicleDetail). If this is too complex, use a simpler approach: snap based on any booked slot across all visible vehicles.
